# Create a map centered around London
m = folium.Map(location=[51.509865, -0.118092], zoom_start=13)

# Add parishes as markers
for parish in parishes:
    folium.Marker(
        location=parish["coordinates"],
        popup=f'<a href="{parish["wikipedia"]}" target="_blank">{parish["name"]}</a>',
        icon=folium.Icon(color="blue", icon="info-sign")
    ).add_to(m)

# Save the map to an HTML file
m.save("parishes_map.html")
